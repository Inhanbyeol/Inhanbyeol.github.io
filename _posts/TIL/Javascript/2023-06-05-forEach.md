> 배열의 요소를 반복하며 함수를 호출한다.
{: .prompt-info }

```js
const test = [1, 3, 5, 7, 9];

test.forEach((value, index, array) => console.log(value));
//1
//3
//5
//7
//9

test.forEach((value, index, array) => console.log(index));
//0
//1
//2
//3
//4

test.forEach((value, index, array) => console.log(array));
//[1, 3, 5, 7, 9]
//[1, 3, 5, 7, 9]
//[1, 3, 5, 7, 9]
//[1, 3, 5, 7, 9]
//[1, 3, 5, 7, 9]
```
